14) const handleSubmit = async (e) => {
    e.preventDefault()
    try {
        const res = await axios.post(`${url}/api/v1/user/signup`, { username, email, password }, { withCredentials: true })
        if (res.data.success) { dispatch({ type: 'auth/setAuthUser', payload: res.data.user }); navigate('/') }
    } catch (error) { console.error(error) }
}
- Prevent default, post signup data, handle success by saving user and navigating to home.
