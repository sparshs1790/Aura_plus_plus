8) useEffect(() => {
    const fetchSuggested = async () => {
        try {
            const res = await axios.get(`${url}/api/v1/user/suggested`, { withCredentials: true })
            if (res.data.success) dispatch(setSuggestedUsers(res.data.users))
        } catch (error) { console.error(error) }
    }
    fetchSuggested()
}, [dispatch])
- On mount, call the suggested users endpoint and dispatch results into Redux.
